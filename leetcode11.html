11. Container With Most Water

Zidong Zhang Jun 22 2019 
    '''

        def maxArea(self, height) -> int:
            biggest = 0
            for i in range(len(height)):
                for j in range(i+1, len(height)):
                    temp = (j - i) * (height[i] if height[i] < height[j] else height[j])
                    biggest = temp if temp > biggest else biggest

    '''
    
    class Solution:
        def maxArea(self, height) -> int:
            length = len(height)
            biggest = 0
            l = 0
            r = len(height) - 1
            hl, hr = 0, 0
            while l < r:
                hl, hr = height[l], height[r]
                temp = min(hl, hr) * (r - l)
                biggest = max(temp, biggest)
                if hl <= hr:
                    l += 1
                else:
                    r -= 1
            return biggest
